
#
#  Branches
#

# Create a new branch and switch to it
git checkout -b <branch name>

# Push a branch that exists on your local to a Remote Repository that does not yet have that branch
git push -u origin <branch name>





#
#  Remote Repositories
#

# View the status of the remotes on your local
git remote -v update

# Update the status of the remotes on your local
git remote update
